3use core::sync::atomic::{AtomicU64, Ordering};
4use handler_table::HandlerTable;
5
6use crate::platform::irq::{MAX_IRQ_COUNT, dispatch_irq};
7use crate::trap::{IRQ, register_trap_handler};
8
9pub use crate::platform::irq::{register_handler, set_enable};
10
11/// The type if an IRQ handler.
12pub type IrqHandler = handler_table::Handler;
13
14static IRQ_HANDLER_TABLE: HandlerTable<MAX_IRQ_COUNT> = HandlerTable::new();
15
16static IRQ_COUNTERS: [AtomicU64; MAX_IRQ_COUNT] = {
17    const INIT: AtomicU64 = AtomicU64::new(0);
18    [INIT; MAX_IRQ_COUNT]
19};
20
21fn increment_irq_count(irq_num: usize) {
22    #[cfg(target_arch = "riscv64")]
23    let actual_irq = if irq_num >= (1 << (usize::BITS - 1)) {
24        irq_num - (1 << (usize::BITS - 1))
25    } else {
26        irq_num
27    };
28
29    #[cfg(not(target_arch = "riscv64"))]
30    let actual_irq = irq_num;
31
32    if actual_irq < MAX_IRQ_COUNT {
33        IRQ_COUNTERS[actual_irq].fetch_add(1, Ordering::Relaxed);
34    }
35}
36
37/// Platform-independent IRQ dispatching.
38#[allow(dead_code)]
39pub(crate) fn dispatch_irq_common(irq_num: usize) {
40    trace!("IRQ {}", irq_num);
41    if !IRQ_HANDLER_TABLE.handle(irq_num) {
42        warn!("Unhandled IRQ {}", irq_num);
43    }
44}
45
46/// Platform-independent IRQ handler registration.
47///
48/// It also enables the IRQ if the registration succeeds. It returns `false` if
49/// the registration failed.
50#[allow(dead_code)]
51pub(crate) fn register_handler_common(irq_num: usize, handler: IrqHandler) -> bool {
52    if irq_num < MAX_IRQ_COUNT && IRQ_HANDLER_TABLE.register_handler(irq_num, handler) {
53        set_enable(irq_num, true);
54        return true;
55    }
56    warn!("register handler for IRQ {} failed", irq_num);
57    false
58}
59
60#[register_trap_handler(IRQ)]
61fn handler_irq(irq_num: usize) -> bool {
62    let guard = kernel_guard::NoPreempt::new();
63    increment_irq_count(irq_num);
64    dispatch_irq(irq_num);
65    drop(guard); // rescheduling may occur when preemption is re-enabled.
66    true
67}
68
69/// Get the count of all IRQs.
70pub fn get_all_irq_counts() -> [u64; MAX_IRQ_COUNT] {
71    let mut counts = [0u64; MAX_IRQ_COUNT];
72    for (i, counter) in IRQ_COUNTERS.iter().enumerate() {
73        counts[i] = counter.load(Ordering::Relaxed);
74    }
75    counts
76}
