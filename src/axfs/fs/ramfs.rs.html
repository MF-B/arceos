5/// `InterruptFile` is a virtual file node that provides IRQ statistics in RAMFS.
6/// path: `/proc/interrupts`
7pub struct InterruptFile;
8
9impl VfsNodeOps for InterruptFile {
10    fn get_attr(&self) -> VfsResult<VfsNodeAttr> {
11        Ok(VfsNodeAttr::new(
12            VfsNodePerm::from_bits_truncate(0o444),
13            VfsNodeType::File,
14            0,
15            0,
16        ))
17    }
18
19    fn read_at(&self, offset: u64, buf: &mut [u8]) -> VfsResult<usize> {
20        let mut output = String::new();
21
22        #[cfg(feature = "irq")]
23        {
24            let counts = axhal::irq::get_all_irq_counts();
25            for (irq_num, count) in counts.iter().enumerate() {
26                if *count > 0 {
27                    output.push_str(&alloc::format!("{irq_num}:\t\t{count}\n"));
28                }
29            }
30        }
31
32        if output.is_empty() {
33            output.push_str("No IRQ activity detected\n");
34        }
35
36        let bytes = output.as_bytes();
37        let available_len = bytes.len().saturating_sub(offset as usize);
38        let copy_len = core::cmp::min(buf.len(), available_len);
39
40        if copy_len > 0 && offset < bytes.len() as u64 {
41            buf[..copy_len].copy_from_slice(&bytes[offset as usize..offset as usize + copy_len]);
42        }
43
44        Ok(copy_len)
45    }
46
47    fn write_at(&self, _offset: u64, _buf: &[u8]) -> VfsResult<usize> {
48        Err(VfsError::PermissionDenied)
49    }
50
51    fn truncate(&self, _size: u64) -> VfsResult {
52        Err(VfsError::Unsupported)
53    }
54
55    axfs_vfs::impl_vfs_non_dir_default! {}
56}
